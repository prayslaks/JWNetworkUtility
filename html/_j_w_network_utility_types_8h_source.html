64inline EJWNU_HttpStatusCode JWNU_IntToHttpStatusCode(const int32 StatusCode)
65{
66 switch (StatusCode)
67 {
68 // 0: UE HTTP 모듈에서 응답 없음 (연결 실패 등)
69 case 0: return EJWNU_HttpStatusCode::NetworkError;
70
71 // 2XX
72 case 200: return EJWNU_HttpStatusCode::OK;
73 case 201: return EJWNU_HttpStatusCode::Created;
74 case 202: return EJWNU_HttpStatusCode::Accepted;
75 case 204: return EJWNU_HttpStatusCode::NoContent;
76
77 // 3XX
78 case 301: return EJWNU_HttpStatusCode::MovedPermanently;
79 case 302: return EJWNU_HttpStatusCode::Found;
80 case 304: return EJWNU_HttpStatusCode::NotModified;
81
82 // 4XX
83 case 400: return EJWNU_HttpStatusCode::BadRequest;
84 case 401: return EJWNU_HttpStatusCode::Unauthorized;
85 case 402: return EJWNU_HttpStatusCode::PaymentRequired;
86 case 403: return EJWNU_HttpStatusCode::Forbidden;
87 case 404: return EJWNU_HttpStatusCode::NotFound;
88 case 405: return EJWNU_HttpStatusCode::MethodNotAllowed;
89 case 406: return EJWNU_HttpStatusCode::NotAcceptable;
90 case 407: return EJWNU_HttpStatusCode::ProxyAuthRequired;
91 case 408: return EJWNU_HttpStatusCode::RequestTimeout;
92 case 409: return EJWNU_HttpStatusCode::Conflict;
93 case 410: return EJWNU_HttpStatusCode::Gone;
94 case 429: return EJWNU_HttpStatusCode::TooManyRequests;
95
96 // 5XX
97 case 500: return EJWNU_HttpStatusCode::InternalServerError;
98 case 501: return EJWNU_HttpStatusCode::NotImplemented;
99 case 502: return EJWNU_HttpStatusCode::BadGateway;
100 case 503: return EJWNU_HttpStatusCode::ServiceUnavailable;
101 case 504: return EJWNU_HttpStatusCode::GatewayTimeout;
102
103 default: return EJWNU_HttpStatusCode::UnknownError;
104 }
105}
