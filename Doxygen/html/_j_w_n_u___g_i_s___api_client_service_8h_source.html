179template <typename StructType>
180void UJWNU_GIS_ApiClientService::CallApi_Template(
181 const UObject* WorldContextObject,
182 const EJWNU_HttpMethod InMethod,
183 const EJWNU_ServiceType InServiceType,
184 const FString InEndpoint,
185 const FString& InContentBody,
186 const TMap<FString, FString>& InQueryParams,
187 TFunction<void(const StructType&)> OnGetCustomStruct,
188 const bool bRequiresAuth)
189{
190 // 객체 획득
191 UJWNU_GIS_ApiClientService* Self = Get(WorldContextObject);
192 if (Self == nullptr)
193 {
194 return;
195 }
196
197 // 호스트 프로바이더에서 호스트 획득
198 FString ProvidedHost;
199 if (const auto HostProvider = UJWNU_GIS_ApiHostProvider::Get(WorldContextObject))
200 {
201 if (HostProvider->GetHost(InServiceType, ProvidedHost) == false)
202 {
203 PRINT_LOG(LogJWNU_GIS_ApiClientService, Warning, TEXT("호스트 획득 실패!"));
204 StructType ErrorResult;
205 ErrorResult.Code = TEXT("HOST_NOT_FOUND");
206 ErrorResult.Message = TEXT("Failed to get host from provider");
207 OnGetCustomStruct(ErrorResult);
208 return;
209 }
210 }
211
212 // API URL 구축
213 FString ConstructedURL = ProvidedHost + InEndpoint;
214
215 // 인증이 필요하지 않은 경우, 토큰 로직을 건너뛰고 바로 실행
216 if (bRequiresAuth == false)
217 {
218 Self->CallApi_Template_Execution(InMethod, InServiceType, ConstructedURL, TEXT(""), InContentBody, InQueryParams, OnGetCustomStruct, false);
219 return;
220 }
221
222 // 토큰 프로바이더에서 엑세스 토큰 획득
223 FJWNU_AccessTokenContainer ProvidedAccessTokenContainer;
224 if (const auto TokenProvider = UJWNU_GIS_ApiIdentityProvider::Get(WorldContextObject))
225 {
226 if (TokenProvider->GetAccessTokenContainer(InServiceType, ProvidedAccessTokenContainer) == false)
227 {
228 PRINT_LOG(LogJWNU_GIS_ApiClientService, Warning, TEXT("엑세스 토큰 획득 실패!"));
229 StructType ErrorResult;
230 ErrorResult.Code = TEXT("TOKEN_NOT_FOUND");
231 ErrorResult.Message = TEXT("Failed to get access token from provider");
232 OnGetCustomStruct(ErrorResult);
233 return;
234 }
235 }
236 else
237 {
238 PRINT_LOG(LogJWNU_GIS_ApiClientService, Warning, TEXT("토큰 프로바이더 획득 실패!"));
239 StructType ErrorResult;
240 ErrorResult.Code = TEXT("PROVIDER_NOT_FOUND");
241 ErrorResult.Message = TEXT("Failed to get token provider");
242 OnGetCustomStruct(ErrorResult);
243 return;
244 }
245
246 // 토큰 만료 또는 리프레시 진행 중인 경우 잡 큐에 적재
247 const int64 CurrentUnixTime = FDateTime::UtcNow().ToUnixTimestamp();
248 const bool bTokenExpired = ProvidedAccessTokenContainer.ExpiresAt > 0 && CurrentUnixTime >= (ProvidedAccessTokenContainer.ExpiresAt - 30);
249 if (bTokenExpired || Self->RefreshInProgressFlags.FindOrAdd(InServiceType))
250 {
251 PRINT_LOG(LogJWNU_GIS_ApiClientService, Display, TEXT("엑세스 토큰 만료 또는 리프레시 진행 중, 잡 큐에 적재..."));
252 JWNU_SCREEN_DEBUG(-1, 5.0f, FColor::Yellow, TEXT("[JWNU] Access Token Expired — Queuing refresh for %s"), *UEnum::GetValueAsString(InServiceType));
253 FJWNU_PendingJob Job;
254 Job.RequestInfo.ServiceType = InServiceType;
255 Job.RequestInfo.Method = InMethod;
256 Job.RequestInfo.URL = ConstructedURL;
257 Job.RequestInfo.ContentBody = InContentBody;
258 Job.RequestInfo.QueryParams = InQueryParams;
259 Job.OnTokenReady = [Self, InMethod, InServiceType, ConstructedURL, InContentBody, InQueryParams, OnGetCustomStruct](const FString& NewAccessToken)
260 {
261 Self->CallApi_Template_Execution<StructType>(InMethod, InServiceType, ConstructedURL, NewAccessToken, InContentBody, InQueryParams, OnGetCustomStruct, false);
262 };
263 Job.OnTokenFailed = [OnGetCustomStruct](const FString& ErrorCode, const FString& ErrorMessage)
264 {
265 StructType ErrorResult;
266 ErrorResult.Code = ErrorCode;
267 ErrorResult.Message = ErrorMessage;
268 OnGetCustomStruct(ErrorResult);
269 };
270 Self->RequestTokenRefresh(InServiceType, MoveTemp(Job));
271 return;
272 }
273
274 // 실제 처리
275 Self->CallApi_Template_Execution(InMethod, InServiceType, ConstructedURL, ProvidedAccessTokenContainer.AccessToken, InContentBody, InQueryParams, OnGetCustomStruct, true);
276}

278template <typename StructType>
279void UJWNU_GIS_ApiClientService::CallApi_Template_Execution(
280 const EJWNU_HttpMethod InMethod,
281 const EJWNU_ServiceType InServiceType,

283 const FString& InAccessToken,
284 const FString& InContentBody,
285 const TMap<FString, FString>& InQueryParams,
286 TFunction<void(const StructType&)> OnGetCustomStruct,
287 const bool bTryTokenRefreshing)

292 FJWNU_PendingApiRequest PendingRequest;
293 PendingRequest.ServiceType = InServiceType;
294 PendingRequest.Method = InMethod;
295 PendingRequest.URL = InURL;

298
299 // 401 상태 코드를 처리할 수 있는 콜백
300 const auto CallbackManage401 = FOnHttpRequestCompletedDelegate::CreateWeakLambda(this,
301 [this, PendingRequest, OnGetCustomStruct](const int32 StatusCode, const FString& ResponseBody)
302 {
303 if (StatusCode == 401)
304 {
305 PRINT_LOG(LogJWNU_GIS_ApiClientService, Display, TEXT("401 토큰 만료 감지, 잡 큐에 적재 후 리프레시 시도..."));
306 JWNU_SCREEN_DEBUG(-1, 5.0f, FColor::Orange, TEXT("[JWNU] 401 Unauthorized — Triggering token refresh for %s"), *UEnum::GetValueAsString(PendingRequest.ServiceType));
307 FJWNU_PendingJob Job;
308 Job.RequestInfo = PendingRequest;
309 Job.OnTokenReady = [this, PendingRequest, OnGetCustomStruct](const FString& NewAccessToken)
310 {
311 CallApi_Template_Execution<StructType>(PendingRequest.Method, PendingRequest.ServiceType, PendingRequest.URL, NewAccessToken, PendingRequest.ContentBody, PendingRequest.QueryParams, OnGetCustomStruct, false);
312 };
313 Job.OnTokenFailed = [OnGetCustomStruct](const FString& ErrorCode, const FString& ErrorMessage)
314 {
315 StructType ErrorResult;
316 ErrorResult.Code = ErrorCode;
317 ErrorResult.Message = ErrorMessage;
318 OnGetCustomStruct(ErrorResult);
319 };
320 RequestTokenRefresh(PendingRequest.ServiceType, MoveTemp(Job));
321 return;
322 }
323
324 // 언리얼 구조체 파싱 시도
325 StructType ResultData;
326 if (FJsonObjectConverter::JsonObjectStringToUStruct(ResponseBody, &ResultData, 0, 0))
327 {
328 OnGetCustomStruct(ResultData);
329 }
330 else
331 {
332 ResultData.Code = TEXT("JSON_PARSE_ERROR");
333 ResultData.Message = TEXT("Failed to Parse JSON Response Body");
334 OnGetCustomStruct(ResultData);
335 }
336 });
337
338 // Http 리퀘스트
339 UJWNU_GIS_HttpClientHelper::SendReqeust_CustomResponse(GetWorld(), InMethod, InURL, InAccessToken, InContentBody, InQueryParams, CallbackManage401);
340 }
341 else
342 {
343 // 401 상태 코드를 처리하지 않는 콜백
344 const auto CallbackNoManage401 = FOnHttpRequestCompletedDelegate::CreateWeakLambda(this,
345 [this, OnGetCustomStruct](const int32 StatusCode, const FString& ResponseBody)
346 {
347 // 언리얼 구조체 파싱 시도
348 StructType ResultData;
349 if (FJsonObjectConverter::JsonObjectStringToUStruct(ResponseBody, &ResultData, 0, 0))
350 {
351 OnGetCustomStruct(ResultData);
352 }
353 else
354 {
355 ResultData.Code = TEXT("JSON_PARSE_ERROR");
356 ResultData.Message = TEXT("Failed to Parse JSON Response Body");
357 OnGetCustomStruct(ResultData);
358 }
359 });
360
361 // Http 리퀘스트
362 UJWNU_GIS_HttpClientHelper::SendReqeust_CustomResponse(GetWorld(), InMethod, InURL, InAccessToken, InContentBody, InQueryParams, CallbackNoManage401);
363 }
